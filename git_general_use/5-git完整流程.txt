提交异常
fatal: remote origin already exists.  

	解决方法
	git remote -v 查看远程库信息
	git remote rm origin 删除远程库列表

#删除.git文件夹中最大的文件
	

	#查询前5个最大的文件
	git verify-pack -v .git/objects/pack/pack-*.idx | sort -k 3 -g | tail -5
	如果没有信息执行如下命令：git gc
	

	#查找文件名称
	git rev-list --objects --all | grep 3ef8dfc43184

	#通过索引删除文件
	git filter-branch --index-filter 'git rm --cached --ignore-unmatch  p0/pycharm-professional-2019.1.3.exe'

	#删除目录
	rm -Rf .git/refs/original .git/logs/

	# 执行删除操作
	git gc --prune=now

git 提交命令
	git add -A  提交所有变化
	git add -u  提交 被修改和被删除文件，新创建文件无法提交。
	git add .   提交新文件和 被修改(modified)文件，被删除文件无法提交
	


.gitignore忽略指定文件
.gitignore 文件用来告诉 git 不应跟踪的文件。该文件应该放在 .git 目录所在的目录。
	
.gitignore文件用来告诉git不跟踪那些文件。
